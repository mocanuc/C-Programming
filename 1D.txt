#include "stdafx.h"
#include<iostream>
#include<vector>
#include <cstdlib>
using namespace std;




int main()

{
	// generates a random vector
	vector<int> v(6);
	cout << "My vector is :" << endl;
	for (int i = 0; i < v.size(); i++)
	{
		v[i]= rand() % 100 +1;
		cout << v[i] << "|";
	}
	
	// shuffle the vector elements based on the random generators
	int temp;
	cout << "My suffled vector is :" << endl;
	for (int i = 0; i < v.size(); i++)
	{
		int randGen = rand() % 100 + 1;
		if (randGen <= 50)
		{
			temp = v.front();
			v.front() = v[i];
			v[i] = temp;
		}
		else
		{
			temp = v.back();
			v.back() = v[i];
			v[i] = temp;
		}
	}
	// printing the results
	for (int i = 0; i < v.size(); i++)
	{
		cout << v[i] << "|";
	}

	return 0;
}
